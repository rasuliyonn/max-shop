Цели и задачи проекта

🎯 Основная цель

Разработать современное веб-приложение для покупки одежды и аксессуаров с интуитивным интерфейсом и полным функционалом электронного магазина.

Целевая аудитория: Покупатели от 18 до 55 лет, предпочитающие покупки в интернете

✅ Задачи

  • ✓ Реализовать каталог товаров с фильтрацией
  • ✓ Добавить функцию поиска
  • ✓ Реализовать корзину покупок
  • ✓ Сделать мобильную версию
  • ✓ Обеспечить быструю загрузку
  • ✓ Создать удобный UI/UX

Zustand - Функция addCart

addCart: (product) => set((state) => { const existing = state.carts.find( (item) => item.id === product.id && item.color === product.color ); if (existing) { // Товар есть - увеличиваем qty return { carts: state.carts.map((item) => item.id === product.id && item.color === product.color ? { ...item, quantity: item.quantity + 1 } : item ), }; } // Новый товар return { carts: [...state.carts, {...product, qty: 1}] }; }),

Zustand - increaseQty и decreaseQty

// Увеличение increaseQty: (id, color) => set((state) => ({ carts: state.carts.map((item) => item.id === id && item.color === color ? { ...item, quantity: item.quantity + 1 } : item ), })), // Уменьшение или удаление decreaseQty: (id, color) => set((state) => { const item = state.carts.find( (i) => i.id === id && i.color === color ); if (item && item.quantity <= 1) { // Удаляем если qty = 1 return { carts: state.carts.filter( (i) => !(i.id === id && i.color === color) ), }; } return { carts: state.carts.map((i) => i.id === id && i.color === color ? { ...i, quantity: i.quantity - 1 } : i ), }; }),

Zustand - Функция deleteCart

deleteCart: (id, color) => set((state) => ({ // Удаляем товар по (id, color) паре carts: state.carts.filter( (item) => !(item.id === id && item.color === color) ), })), // Используется при клике "✕" в корзине // После удаления: // - total пересчитывается // - Basket компонент обновляется // - если пусто → сообщение

Persist Middleware - Сохранение

export const useCartStore = create( persist( (set) => ({ /* store */ }), { name: "cart-storage", partialize: (state) => ({ carts: state.carts // Сохраняем корзину }), merge: (persistedState, currentState) => ({ ...currentState, ...(persistedState || {}), product: products, // Свежие товары }), }, ), ); // Result: localStorage["cart-storage"] = JSON
